\begin{solusi}
    Perhatikan bahwa $a = (0*0)*a$. Berarti 
    \begin{align*}
        (0*0)+b+c=((0*0)*b)*c=b*c
    \end{align*}
    sehingga 
    \begin{align*}
        (0*0)+(0*0)+c=(0*0)*c=c \implies (0*0)+(0*0) = 0 \implies 0*0 = 0.
    \end{align*}
    Oleh karena itu didapat
    \begin{align*}
        a*b = (0*0)+a+b = 0+a+b=a+b
    \end{align*}
    terbukti.
\end{solusi}
